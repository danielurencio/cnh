SELECT A.RONDA,A.LICITACION,A.VAR_ADJ1,
A.VAR_ADJ2,A.VPO,A.BONO,A.ID_LICITANTE,
A.ID_BLOQUE,B.ID_MATRIZ AS ID_EMPRESA,
C.ID_LICITANTE_ADJ,D.MATRIZ AS EMPRESA,
D.PAIS, C.INV_USD, C.SUPERFICIE AS AREA,
A.VALIDEZ,E.LICITANTE,C.POZOS_COMPROMETIDOS,C.CONTRATO
FROM DATOS_LICITACIONES_OFERTAS_PRUEBAS A
LEFT JOIN DATOS_LICITACIONES_LIC_MAT B
ON A.ID_LICITANTE = B.ID_LICITANTE
LEFT JOIN DATOS_LICITACIONES_BLOQUES C
ON A.ID_BLOQUE = C.ID_BLOQUE
LEFT JOIN DATOS_LICITACIONES_MATRICES D
ON D.ID_MATRIZ = B.ID_MATRIZ
LEFT JOIN DATOS_LICITACIONES_LICITANTES E
ON A.ID_LICITANTE = E.ID_LICITANTE;
