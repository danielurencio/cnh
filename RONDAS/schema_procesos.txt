CREATE TABLE DATOS_LICITACIONES_PROCESOS(
  ID_LICITANTE NUMBER(8,0),
  MOD VARCHAR2(100),
  ID_R VARCHAR2(100),
  RONDA VARCHAR2(100), 
  LIC VARCHAR2(100),

);
