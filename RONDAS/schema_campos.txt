CREATE TABLE DATOS_LICITACIONES_CAMPOS(
 ID_BLOQUE VARCHAR2(100),
 CAMPO VARCHAR2(100),
 VOL_ORIGINAL NUMBER(38,11),
 VOL_ORIGINAL_ACEITE NUMBER(38,11),
 VOL_ORIGINAL_GAS NUMBER(38,11),
 RESERVAS_1P NUMBER(38,12),
 RESERVAS_2P NUMBER(38,12),
 RESERVAS_3P NUMBER(38,12),
 CRUDO_1P NUMBER(38,12),
 CRUDO_2P NUMBER(38,12),
 CRUDO_3P NUMBER(38,12),
 GAS_1P NUMBER(38,12),
 GAS_2P NUMBER(38,12),
 GAS_3P NUMBER(38,12),
 Foreign key (ID_BLOQUE) references DATOS_LICITACIONES_BLOQUES(ID_BLOQUE)
);
