CREATE TABLE RONDAS_LICITANTES(
 ID_LICITANTE NUMBER(8,0),
 ID VARCHAR2(100),
 RONDA VARCHAR2(100),
 LIC VARCHAR2(100),
 BLOQUE VARCHAR2(100),
 NUM VARCHAR2(100),
 EMPRESA VARCHAR2(200),

 OPERADOR VARCHAR2(200),

);
