CREATE TABLE DATOS_LICITACIONES_OFERTAS(
  ID_LICITACION VARCHAR2(100),
  RONDA VARCHAR2(100),
  LICITACION VARCHAR2(100),
  NUM_BLOQUE VARCHAR2(100),
  MIN_VAR_ADJ1 NUMBER(34,1),
  MAX_VAR_ADJ1 NUMBER(34,0),
  VAR_ADJ1 NUMBER(34,2),
  VAR_ADJ2 NUMBER(34,0),
  VPO NUMBER(35,3),
  VALIDEZ VARCHAR2(100),
  BONO NUMBER(35,2),
  INSACULACION VARCHAR2(100),
  POSICION NUMBER(35,0),
  ID_LICITANTE NUMBER(34,0),
  Foreign key (ID_LICITANTE) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE)
);
