CREATE TABLE DATOS_LICITACIONES_OFERTAS(
 ID_LICITANTE_OFERTA NUMBER(8,0),
 ID_BLOQUE VARCHAR2(100),
 VAR_ADJ1 NUMBER(28,22),
 VAR_ADJ2 NUMBER(28,22),
 VPO NUMBER(35,22),
 BONO NUMBER(38,22),
 ID_OPERADOR NUMBER(8,0),
 GANADOR NUMBER(8,0),
 SEGUNDO_LUGAR NUMBER(8,0),
 ID_ADJ NUMBER(8,0),
 ID_LICITANTE_ADJ NUMBER(8,0),
 Foreign key (ID_LICITANTE_OFERTA) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (GANADOR) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (SEGUNDO_LUGAR) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (ID_LICITANTE_ADJ) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (ID_OPERADOR) references DATOS_LICITACIONES_OPERADORES(ID_OPERADOR)
);
