/*CREATE TABLE OFERTAS_2 AS*/
SELECT A.*,B.ID_MATRIZ,C.INV_USD,D.AREA,C.RONDA,C.LICITACION
FROM TABLA_ A
LEFT JOIN DATOS_LICITACIONES_LIC_MAT B
ON A.ID_LICITANTE = B.ID_LICITANTE
LEFT JOIN DATOS_LICITACIONES_BLOQUES1 C
ON A.ID_BLOQUE = C.ID_BLOQUE1
LEFT JOIN DATOS_LICITACIONES_BLOQUES D
ON C.ID_BLOQUE1 = D.ID_BLOQUE;
