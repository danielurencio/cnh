/*CREATE TABLE LICRONDAS_2 AS*/
SELECT A.RONDA, A.LICITACION, B.ID_LICITANTE_ADJ, A.ID_LICITANTE, C.LICITANTE
FROM DATOS_LICITACIONES_OFERTAS1 A
LEFT JOIN DATOS_LICITACIONES_BLOQUES1 B
ON A.ID_BLOQUE = ID_BLOQUE1
LEFT JOIN DATOS_LICITACIONES_LICITANTES1 C
ON A.ID_LICITANTE = C.ID_LICITANTE
WHERE A.VALIDEZ = 'VALIDA';
