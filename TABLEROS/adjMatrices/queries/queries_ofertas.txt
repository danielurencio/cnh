SELECT A.*, B.PMT_TOTAL, B.RONDA, B.LICITACION, B.AREA, C.ID_EMPRESA
FROM DATOS_LICITACIONES_OFERTAS A
LEFT JOIN DATOS_LICITACIONES_BLOQUES B
ON A.ID_BLOQUE = B.ID_BLOQUE
INNER JOIN DATOS_LICITACIONES_LIC_EMP C
ON C.ID_LICITANTE = A.GANADOR;
