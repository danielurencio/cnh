CREATE TABLE T0 AS
SELECT ID_LICITANTE_ADJ, ID_BLOQUE
FROM DATOS_LICITACIONES_OFERTAS
WHERE ID_LICITANTE_OFERTA=ID_LICITANTE_ADJ;

CREATE TABLE T1 AS
SELECT A.*, B.AREA, B.INV_TOTAL
FROM T0 A
LEFT JOIN DATOS_LICITACIONES_BLOQUES B
ON A.ID_BLOQUE = B.ID_BLOQUE;

CREATE TABLE T2 AS
SELECT ID_LICITANTE_ADJ, COUNT(ID_BLOQUE) AS ADJUDICADOS, SUM(AREA) AS AREA, SUM(INV_TOTAL) AS INV_TOTAL
FROM T1
GROUP BY ID_LICITANTE_ADJ;>

CREATE TABLE T3 AS
SELECT A.*, B.*
FROM T2 A
LEFT JOIN DATOS_LICITACIONES_LIC_EMP B
ON B.ID_LICITANTE = A.ID_LICITANTE_ADJ;

CREATE TABLE T4 AS
SELECT A.EMPRESA,A.PAIS,B.*
FROM T3 B
LEFT JOIN DATOS_LICITACIONES_EMPRESAS A
ON B.ID_EMPRESA=A.ID_EMPRESA;

CREATE TABLE T5 AS
SELECT A.*, B.MODALIDAD
FROM T4 A
LEFT JOIN DATOS_LICITACIONES_LICITANTES B
ON A.ID_LICITANTE = B.ID_LICITANTE;
