CREATE TABLE TEMP AS
SELECT A.ID_LICITANTE_OFERTA, A.ID_LICITANTE_ADJ, A.ID_BLOQUE, B.RONDA, B.LICITACION
FROM DATOS_LICITACIONES_OFERTAS A
LEFT JOIN DATOS_LICITACIONES_BLOQUES B
ON A.ID_BLOQUE = B.ID_BLOQUE;

SELECT A.*, B.LICITANTE
FROM TEMP A
LEFT JOIN DATOS_LICITACIONES_LICITANTES B
ON A.ID_LICITANTE_OFERTA = B.ID_LICITANTE
