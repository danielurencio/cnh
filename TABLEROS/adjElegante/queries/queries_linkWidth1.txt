/* Â¿la buena? */
create table licRondas0 as
select ID_BLOQUE, RONDA, LICITACION
FROM DATOS_LICITACIONES_BLOQUES;

CREATE TABLE licRondas1 as
SELECT A.*, B.ID_LICITANTE_ADJ, B.ID_LICITANTE_OFERTA
FROM licRondas0 A
LEFT JOIN DATOS_LICITACIONES_OFERTAS B
ON A.ID_BLOQUE = B.ID_BLOQUE;

CREATE TABLE licRondas2 as
SELECT A.*, B.LICITANTE
FROM licRondas1 A
LEFT JOIN DATOS_LICITACIONES_LICITANTES B
ON A.ID_LICITANTE_OFERTA = B.ID_LICITANTE;
