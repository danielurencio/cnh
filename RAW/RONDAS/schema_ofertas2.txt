CREATE TABLE DATOS_LICITACIONES_OFERTAS(
 ID_LICITANTE NUMBER(8,0),
 ID_RONDA VARCHAR2(100),
 RONDA VARCHAR2(100),
 LIC VARCHAR2(100),
 BLOQUE VARCHAR2(100),
 NUM VARCHAR2(100),
 CONTRATO VARCHAR2(100),
 VAR_ADJ1 NUMBER(28,22),
 VAR_ADJ2 NUMBER(28,22),
 VPO NUMBER(28,22),
 BONO NUMBER(38,22),
 OPERADOR NUMBER(8,0),
 GANADOR NUMBER(8,0),
 SEGUNDO_LUGAR NUMBER(8,0),
 ID_ADJ NUMBER(8,0),
 ADJ NUMBER(8,0),
 Foreign key (ID_LICITANTE) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (GANADOR) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (SEGUNDO_LUGAR) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (ID_ADJ) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (ADJ) references DATOS_LICITACIONES_LICITANTES(ID_LICITANTE),
 Foreign key (OPERADOR) references DATOS_LICITACIONES_OPERADORES(ID_OPERADOR)
);
